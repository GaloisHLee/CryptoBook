unsigned char Division(unsigned short Num_L, unsigned short Num_R, unsigned short *Remainder)
{
unsigned short r0 = 0;
unsigned char q = 0;
int bitCount = 0;

r0 = Num_L;

bitCount = GetHighestPosition(r0) - GetHighestPosition(Num_R);
while (bitCount >= 0)
{
q = q | (1 << bitCount);
r0 = r0 ^ (Num_R << bitCount);
bitCount = GetHighestPosition(r0) - GetHighestPosition(Num_R);
}
*Remainder = r0;
return q;
}
